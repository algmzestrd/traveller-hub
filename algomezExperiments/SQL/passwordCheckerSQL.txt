<?
function checkPass($user, $password)
{
	
        $query = "SELECT UserID, NetID, Passwd
	          FROM interview_user_info
		  WHERE NetID=";
		  
        $query .= "'" . $user . "'";
        
	$result = mysql_query($query);
	
	$array = mysql_fetch_assoc($result);
	
	$hash = md5($password);
	
	if($hash == $array["Passwd"])
	    $UserID = $array["UserID"];
	else
	    $UserID = -1;


	return $UserID;

}
?>
